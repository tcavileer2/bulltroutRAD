# http://catchenlab.life.illinois.edu/stacks/manual/#intro Run gstacks
# to build loci from the aligned paired-end data. We have instructed
# gstacks to remove any PCR duplicates that it finds.
# #
mkdir 05-Stacks gstacks -I ./04-Mapped/ -M
./fix_metadata/popmap1_LC_location.txt --rm-pcr-duplicates -O
05-Stacks/popmap1_LC_location -t 55

# Run populations. Calculate Hardy-Weinberg deviation, population
# statistics, f-statistics and smooth the statistics across the genome.
# Export several output files.
# 
populations -P 05-Stacks/popmap1_LC_location -M
./fix_metadata/popmap1_LC_location.txt -r 0.65 --ordered_export --vcf
--genepop --structure --radpainter --phase --fastphase --plink --hzar
--phylip --phylip_var --treemix --fstats --smooth --hwe -t 8
